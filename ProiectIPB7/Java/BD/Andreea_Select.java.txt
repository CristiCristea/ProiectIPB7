import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;

public class BDOperation {

	public String results =new String();
	
	public Connection connect () throws SQLException, ClassNotFoundException
	{
		
			Class.forName("oracle.jdbc.driver.OracleDriver");
			Connection con=DriverManager.getConnection(  
				"jdbc:oracle:thin:@localhost:1521:xe","STUDENT","STUDENT");
			return con;	
		  
	}
	
        public void getProfesori() 
	{
		try
 		   {
			Connection con = this.connect(); //conectarea la BD
			Statement stmt=con.createStatement();
			ResultSet result=stmt.executeQuery("select nume, prenume, grad_didactic from profesori"); 
			while(result.next())  
				this.results = results+result.getString(1)+"\n"+result.getString(2)+"\n"+result.getString(3)+"\n";
			con.close();
		    }
		catch (ClassNotFoundException| SQLException e)
		{
			this.results = "Eroare!";
		}
	}
}
